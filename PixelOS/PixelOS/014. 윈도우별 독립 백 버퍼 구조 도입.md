# 변경 사항

## 코드

**window.h**

```C
...

// Window 구조체 타입 정의
typedef struct Window
{
    ...
	
    // window 전용 백 버퍼
    uint8_t* buffer;
    // 논리적인 window 너비
    int stride;
} Window;

...

// window 버퍼에 텍스트를 출력하는 함수
void window_draw_char(Window* w, int x, int y, char c, uint8_t color);
void window_draw_string(Window* w, int x, int y, const char* s, uint8_t color);

...

// 합성 출력 함수
void wm_composite(void);
```

윈도우 구조체의 멤버 변수로 **윈도우 전용 백 버퍼의 주소를 저장**하는 `buffer`와, **버퍼에서 다음 행으로 이동하기 위한 간격을 의미**하는 `stride`를 추가하였습니다.

**윈도우의 백 버퍼에 문자 및 문자열을 출력**하는 `window_draw_char` 함수와 `window_draw_string` 함수를 선언하였습니다.

**화면 프레임임 백 버퍼에 윈도우 백 버퍼를 z-index 순서로 합성**하는 `wm_composite` 함수를 선언하였습니다.


**window.c**

```C
...

// 윈도우 버퍼
static uint8_t g_window_buffers[MAX_WINDOWS][MAX_W * MAX_H];

// window manager 초기화
void wm_init(void)
{
    // 모든 window를 비활성 상태로 설정
    for (int i = 0; i < MAX_WINDOWS; ++i)
    {
        g_windows[i].in_use = false;
        g_order[i] = 0;
        g_windows[i].buffer = 0;
    }
    g_order_count = 0;
}

// 윈도우 백 버퍼에 픽셀을 그리는 내부 인라인 함수
static inline void win_putpixel(Window* w, int x, int y, uint8_t c)
{
    if (x < 0 || x >= w->width || y < 0 || y >= w->height)
        return;
		
    w->buffer[y * w->stride + x] = c;
}

// 윈도우 백 버퍼에 프레임을 그리는 함수
static void window_draw_frame(Window* w)
{
    // 배경
    for (int y = 0; y < w->height; y++)
        for (int x = 0; x < w->width; x++)
            w->buffer[y * w->stride + x] = w->bg_color;
			
    // 테두리
    for (int x = 0; x < w->width; x++) {
        win_putpixel(w, x, 0, w->border_color);
        win_putpixel(w, x, w->height - 1, w->border_color);
    }
	
    for (int y = 0; y < w->height; y++) {
        win_putpixel(w, 0, y, w->border_color);
        win_putpixel(w, w->width - 1, y, w->border_color);
    }
	
    // 제목 바
    int title_h = FONT_HEIGHT + 2;
    for (int y = 1; y <= title_h; y++)
        for (int x = 1; x < w->width - 1; x++)
            win_putpixel(w, x, y, w->border_color);
			
    // 제목 텍스트
    if (w->title) {
        window_draw_string(w, 4, 2, w->title, COLOR_WHITE);
    }
	
}

// 새 window 생성
Window* wm_create_window(int px, int py, int width, int height,
                         uint8_t bg_color, uint8_t border_color,
                         const char* title)
{
    for (int i = 0; i < MAX_WINDOWS; ++i)
    {
        // 비활성 상태인 window가 존재하는 경우
        if (!g_windows[i].in_use)
        {
			...
			
            // 윈도우 백 버퍼 연결
            win->buffer = g_window_buffers[i];
            win->stride = width;
			
            // 프레임 초기 렌더링
            window_draw_frame(win);
			
            // 생성한 window 반환
            return win;
        }
    }
    // window 생성 실패
    return 0;
}

...

// 정렬/정규화 파이프라인
static void wm_refresh_order()
{
    wm_build_order_list();
    wm_sort_by_zindex();
    wm_normalize_zindex();
}

// 윈도우 버퍼에 문자 출력 함수
void window_draw_char(Window* w, int x, int y, char c, uint8_t color)
{
    if (!w || !w->in_use)
    {
        return;
    }
	
    // ASCII 범위 체크
    if ((unsigned char)c >= 128)
    {
        return;
    }
	
    for (int row = 0; row < FONT_HEIGHT; ++row)
    {
        int py = y + row;
        if (py < 0 || py >= w->height)
        {
            continue;
        }
		
        uint8_t bits = font8x16[(unsigned char)c][row];
		
        for (int col = 0; col < FONT_WIDTH; ++col)
        {
            if (bits & (1 << (7 - col)))
            {
                int px = x + col;
                if (px < 0 || px >= w->width)
                {
                    continue;
                }
				
                w->buffer[py * w->stride + px] = color;
            }
        }
    }
}

void window_draw_string(Window* w, int x, int y, const char* str, uint8_t color)
{
    if (!w || !w->in_use || !str)
    {
        return;
    }
	
    int px = x;
	
    for (const char* s = str; *s; ++s)
    {
        window_draw_char(w, px, y, *s, color);
        px += FONT_WIDTH;
    }
}

...

void wm_composite(void)
{
    // desktop 배경
    gfx_clear(COLOR_LIGHT_GRAY);
	
    wm_refresh_order();
	
    for (int i = 0; i < g_order_count; i++)
    {
        Window* w = g_order[i];
        if (!w->in_use) continue;
		
        for (int y = 0; y < w->height; y++)
        {
            int sy = w->py + y;
            if (sy < 0 || sy >= HEIGHT) continue;
			
            for (int x = 0; x < w->width; x++)
            {
                int sx = w->px + x;
                if (sx < 0 || sx >= WIDTH) continue;
				
                uint8_t c = w->buffer[y * w->stride + x];
                gfx_putpixel(sx, sy, c);
            }
        }
    }
}
```

**각 윈도우가 윈도우 내부 렌더링을 화면과 분리된 백 버퍼에서 수행**할 수 있도록 구성하였습니다.

**윈도우 백 버퍼에 픽셀을 출력**하기 위한 내부 인라인 함수인 `win_putpixel`을 정의하였으며, `(x, y)` 좌표를 `y * stride + x` 인덱스로 변환하여 **윈도우 백 버퍼에 직접 픽셀을 기록**하도록 구현하였습니다. 이때 윈도우 영역을 벗어나는 좌표는 그리지 않도록 설정하였습니다.

**배경 색 채우기, 타이틀 바/테두리 렌더링**을 윈도우의 백 버퍼에 수행하는 로직을 분리하여 `window_draw_frame` 함수로 모듈화하였습니다.

`wm_create_window` 함수에서는 새로 생성된 윈도우에 전용 백 버퍼를 연결하고. 버퍼에서 다음 행으로 이동하기 위한 간격(`stride`)을 설정하도록 수정하였습니다.

`window_draw_char` 및 `window_draw_string` 함수에서는 폰트 비트맵을 기반으로 좌표에 해당하는 픽셀을 계산하여 **윈도우 백 버퍼에 텍스트를 렌더링**하도록 구현하였습니다.

`wm_composite` 함수에서는 **데스크탑 배경을 밝은 회색으로 초기화**한 후, **z-index에 따라 정렬된 윈도우 순서대로 각 윈도우의 백 버퍼를 순회**하면서 **화면 프레임 백 버퍼에 픽셀 단위로 합성** 하는 작업을 수행하도록 구현하였습니다.

# 이슈 및 버그

## 증상

 - 윈도우 내부에 출력한 텍스트 및 프레임이 **수직 방향으로 어긋난 형태로 표시되는 현상** 발생

## 재현

 - 윈도우 생성 시 `stride`를 고정값 `MAX_W`로 설정

## 원인(추정)

 - 윈도우 백 버퍼는 `g_window_buffers[MAX_WINDOWS][MAX_W * MAX_H]`로 **최대 크기 기준으로 할당**되어 있어 stride를 `MAX_W`로 설정하였으나, 실제 픽셀 접근 시 **윈도우 크기만큼의 메모리만을 사용**하는 구조로 설정되어 있음.
 - 즉, **버퍼의 실제 사용 방식과 픽셀 주소 계산 방식이 불일치**한 것이 원인으로 추정됨.

## 해결 및 우회 방법

 - 윈도우 백 버퍼를 **윈도우 크기만큼만 사용하는 현재 구조에 맞추어** `stride` 값을 `MAX_W`가 아닌 **윈도우의 실제 너비(`width`)로 설정**하도록 수정하여 해결하였음.
