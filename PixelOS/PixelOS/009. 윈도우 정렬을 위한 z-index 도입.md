## 코드

**window.h**

```C
...

// z-index 관리를 위한 전역 변수 설정
static int g_top_zindex = 0;

// Window 구조체 타입 정의
typedef struct Window
{
	...
	
    // 윈도우 순서 설정을 위한 z-index
    int z_index;
	
    ...
} Window;

...

// window를 최상위 레이어로 이동시키는 함수

```

**윈도우 구조체 멤버 변수로 `z_index`를 추가**하였으며, **window를 최상위 레이어로 이동시키는 함수를 선언**하였습니다.

**window.c**

```C
...

// 새 window 생성
Window* wm_create_window(int px, int py, int width, int height, uint8_t bg_color, uint8_t border_color, const char* title)
{
    for (int i = 0; i < MAX_WINDOWS; ++i)
    {
        // 비활성 상태인 window가 존재하는 경우
        if (!g_windows[i].in_use)
        {
            // 새로운 window 구조체 생성
            Window* win = &g_windows[i];
            // window 정보 설정
            ...
            
            // 새로 생성된 window를 가장 상위 레이어로 자동 배치
            win->z_index = g_top_zindex++;
            
            // 생성한 window 반환
            return win;
        }

        // 모든 window가 활성 상태인 경우
        else
        {
            // window 생성 실패
            return 0;
        }
    }
}

...

// z-index의 오버플로우 방지를 위한 z-index 일반화 함수
static void wm_normalize_zindex()
{
    int new_index = 0;

    for (int i = 0; i < MAX_WINDOWS; i++)
    {
        if (g_windows[i].in_use)
        {
            g_windows[i].z_index = new_index++;
        }
    }

    g_top_zindex = new_index;
}

// z-index 기준으로 window를 정렬하기 위한 내부 함수 정의
// Insert sort 사용
static void wm_sort_by_zindex()
{
    for (int i = 1; i < MAX_WINDOWS; ++i)
    {
        // 현재 삽입될 윈도우를 key로 설정
        Window key = g_windows[i];
		
        // 비활성 window는 무시
        if (!key.in_use)
        {
            continue;
        }
		
        // Insert Sort 알고리즘을 통해 g_windows를 z-index 기반으로 정렬
        int j = i - 1;
		
        while (j >= 0 && g_windows[j].in_use && 
	        g_windows[j].z_index > key.z_index)
        {
            g_windows[j + 1] = g_windows[j];
            --j;
        }
		
        g_windows[j + 1] = key;
    }
	
    // 오버플로우 방지를 위한 z-index 일반화
    wm_normalize_zindex();
}

// 모든 window를 포함한 화면 재출력 함수
void wm_draw_all(void)
{
    // 전체 배경 색 초기화
    gfx_clear(COLOR_LIGHT_GRAY);
	
    // 출력 전 window를 z-index 기준으로 정렬
    wm_sort_by_zindex();
	
    // 활성 상태인 window 출력
    for (int i = 0; i < MAX_WINDOWS; ++i)
    {
        if (g_windows[i].in_use)
        {
            draw_window(&g_windows[i]);
        }
    }
}

...

// 창을 가장 위 레이어로 가져오는 함수
void wm_bring_to_front(Window* win)
{
    win->z_index = g_top_zindex++;
    wm_sort_by_zindex();
}

```

기존에는 윈도우 생성 순서대로 그려졌으나, `z_index`를 추가하여 윈도우의 렌더링 우선순위를 결정하도록 수정하였습니다.

- `wm_draw_all()`에서 출력 전 `wm_sort_by_zindex()` 함수를 통해 `z_index`를 기준으로 윈도우를 정렬 후, `z_index` 값이 낮은 윈도우부터 렌더링하도록 수정하였습니다.

- `wm_normalize_zindex()`를 통해 활성 창들의 `z_index`를 재 정렬 및 정규화하여, `z_index` 값 증가로 인한 오버플로우를 방지하였습니다.

- `wm_bring_to_front()` 함수를 추가하여 특정 윈도우의 `z_index`를 최상단으로 갱신하여 **맨 앞으로 가져오는 기능**을 제공하도록 구현하였습니다.
