**Ring3에서 Ring0으로 전환되는 시점의 레지스터 상태를 담는 구조체**(`TrapFrame`)를 추가하였습니다.

시스템 콜 핸들러가 해당 구조체를 인자로 전달받고 반환하도록 인터페이스를 변경하였습니다.

```C
void task_init(void)

// 유저 모드 전용 협력 스케줄링 함수
TrapFrame* task_yield_from_user(TrapFrame* frame)
{
    // 전달받은 Trap frame 포인터가 유효하지 않은 경우 무시
    if (!frame)
    {
        return frame;
    }
	
    // 현재 태스크 번호가 범위 밖인 경우 무시
    if (g_current < 0 || g_current >= g_count)
    {
        return frame;
    }
	
    // 현재 상태 스냅샷
    Task* current = &g_tasks[g_current];
    current->is_user = true;
    current->user_frame = frame;
	
    // 실행 중인 태스크가 1개인 경우 전환 생략
    if (g_count <= 1)
    {
        return current->user_frame;
    }
	
    // 다음에 실행할 태스크 탐색
    int next = pick_next(g_current);
    // 없거나 자기 자신인 경우 전환 생략
    if (next < 0 || next == g_current)
    {
        return current->user_frame;
    }
	
    int prev = g_current;
    g_current = next;
    g_current_task = &g_tasks[g_current];
	
    // 전환할 태스크의 전용 커널 스택 Top 주소로 TSS.esp0 갱신
    gdt_set_kernel_stack(task_kernel_stack_top(g_current_task));
	
    // 해당 태스크로 컨텍스트 스위칭
    task_switch(&g_tasks[prev].esp, g_tasks[g_current].esp);
	
    // 유저 모드 태스크가 다시 스케줄링되었을 때 이어서 실행
    if (g_current_task &&
	    g_current_task->is_user &&
	    g_current_task->user_frame
	)
    {
        return g_current_task->user_frame;
    }
	
    return frame;
}

...
```

태스크 구조체(`Task`)에 다음 멤버를 추가하였습니다.
	**ring3 실행 태스크 여부(`is_user`)
	마지막으로 저장한 사용자 Trap frame(`user_frame`)**

**유저 모드 전용 협력 스케줄링 함수(`task_yield_from_user`)를 구현**하였습니다.
	현재 유저 태스크의 레지스터 상태(`TrapFrame`)를 보관하고, **`TSS.esp0`를 전환할 태스크의 커널 스택으로 갱신**하여 태스크를 전환합니다.
	이후 해당 태스크가 다시 스케줄링되었을 때, **저장해두었던 유저 모드 컨텍스트(`user_frame`)를 반환**하도록 하여 `iretd`를 통한 정상 복귀를 보장하였습니다.


**idt.asm**

```ASM
...

;-------------------------------------------------------------------
; 시스템 콜 인터럽트 서비스 루틴
;-------------------------------------------------------------------
syscall_isr:
    pusha
	
    mov esi, esp                ; Trap Frame 사작 주소 (frame)
    mov eax, [esp + 28]         ; 시스템 콜 번호 (syscall_num)
    mov ecx, [esp + 24]         ; 시스템 콜 인자 (arg0)
    mov edx, [esp + 20]         ; 시스템 콜 인자 (arg1)
    mov ebx, [esp + 16]         ; 시스템 콜 인자 (arg2)
	
    push ebx                    ; 5번째 인자: arg2
    push edx                    ; 4번째 인자: arg1
    push ecx                    ; 3번째 인자: arg0
    push eax                    ; 2번째 인자: syscall_num
    push esi                    ; 1번째 인자: TrapFrame* frame
	
    call syscall_handler        ; syscall_handler 호출
	
    add esp, 20                 ; 인자 정리 (스택에서 20바이트 제거)
	
    mov esp, eax
	
    popa
	
    iretd
```

변경된 시스템 콜 핸들러(`syscall_handler`)의 인터페이스에 맞추어 **시스템 콜 ISR을 수정**하였습니다.

`pusha` 직후 `ESP`는 `EDI`를 가리키므로, **`TrapFrame` 구조체의 인스턴스 주소(`&frame`)를 얻기 위해 `ESP` 주소값을 전달**하였습니다.
