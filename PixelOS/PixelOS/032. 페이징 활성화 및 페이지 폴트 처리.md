**paging.c**

```C
...

// CR3 레지스터에 값을 쓰는 내부 헬퍼
static inline void write_cr3(uint32_t value)
{
    // 내부 어셈블리 실행 및 삭제 금지 선언
    __asm__ volatile ("mov %0, %%cr3" :: "r"(value) : "memory");
}

// CR0 레지스터 값을 읽어오는 내부 헬퍼
static inline uint32_t read_cr0(void)
{
    uint32_t value;
    // 내부 어셈블리 실행 및 삭제 금지 선언
    __asm__ volatile ("mov %%cr0, %0" : "=r"(value));
	
    return value;
}

// CR0 레지스터에 값을 쓰는 내부 헬퍼
static inline void write_cr0(uint32_t value)
{
    // 내부 어셈블리 실행 및 삭제 금지 선언
    __asm__ volatile ("mov %0, %%cr0" :: "r"(value) : "memory");
}

...

// 페이징 활성화 함수 (CR3 로드 후 CR0.PE 및 CR0.PG 세팅)
// 유효한 페이지 테이블 보장을 위해 paging_init() 선행 필요
void paging_enable(void)
{
    // 페이지 디렉토리 주소를 CR3 레지스터에 전달
    uint32_t pd_phys = paging_get_directory_phys();
    write_cr3(pd_phys);
	
    // CR0의 페이징 비트(PG) 및 보호 비트(PE) 설정
    uint32_t cr0 = read_cr0();
    cr0 |= 0x80000001u;
    write_cr0(cr0);
}

// 현재 페이지 디렉토리 물리 주소를 반환하는 함수
uint32_t paging_get_directory_phys(void)
{
    return (uint32_t)g_page_directory;
}

...
```

**페이징 활성화를 수행하는 함수**(`paging_enable()`)을 구현하였습니다.
페이징 활성화 과정은 다음과 같습니다.
1. **페이지 디렉토리 주소**를 **CR3 레지스터**에 전달합니다.
2. **CR0 레지스터**의 **페이징 비트(비트 0)**와 **보호 비트(비트 31)**를 **1로 설정**합니다.


**idt.asm**

```Nasm
...

global page_fault_isr

...

extern page_fault_handler

...

;------------------------------------------------------------------
; 페이지 폴트 인터럽트 서비스 루틴
;------------------------------------------------------------------
page_fault_isr:
    pusha
	
    ; 두 번째 인자(error_code) 가져오기
    mov eax, [esp + 32]
    push eax
	
    ; 첫 번째 인자(fault_addr) 가져오기
    mov eax, cr2
    push eax
	
    call page_fault_handler     ; page_fault_handler 호출
    add esp, 8                  ; 인자 정리 (스택에서 8바이트 제거)
	
    popa
	
    add esp, 4                  ; CPU가 스택에 추가한 error_code 정리
	
    iretd
```

**페이지 폴트 인터럽트 서비스 루틴**을 작성하였습니다.

**페이지 폴트 발생 시 CPU가 자동으로 스택에 에러 코드를 푸시하는 특성을 반영**하여 인자 추출 및 실행 흐름 복구 시 스택 포인터를 보정하도록 구현하였습니다.

페이지 폴트 인터럽트 서비스 루틴에서는 **페이지 폴트가 발생한 주소와 에러 코드**를 인자로 전달하여 **페이지 폴트 핸들러를 호출**합니다.

**interrupt.c**

```C
...

extern void page_fault_isr(void);

...

// 페이지 폴트 디버깅용
volatile uint32_t g_last_pf_addr = 0;
volatile uint32_t g_last_pf_err = 0;

...

// 32비트 비부호 정수를 16진수 문자열로 변환하는 내부 헬퍼 (에러 코드 출력용)
static void hex32_to_str(uint32_t v, char out[11])
{
    static const char* hex = "0123456789ABCDEF";
    out[0] = '0';
    out[1] = 'x';
    for (int i = 0; i < 8; ++i)
    {
        uint32_t shift = (7 - i) * 4;
        out[2 + i] = hex[(v >> shift) & 0xF];
    }
    out[10] = 0;
}

// asm 코드에서 호출되는 페이지 폴트 핸들러 정의
void page_fault_handler(uint32_t fault_addr, uint32_t error_code)
{
    g_last_pf_addr = fault_addr;
    g_last_pf_err = error_code;
	
    // 실행 중단
    __asm__ volatile ("cli");
	
    // 화면 초기화(빨간색)
    gfx_clear(COLOR_LIGHT_RED);
	
    // 에러 화면 구성
    put_string(8, 8,  "PAGE FAULT OUTBREAK!", COLOR_WHITE);
    put_string(8, 28, "fault_addr =", COLOR_WHITE);
    put_string(8, 48, "error_code =", COLOR_WHITE);
	
    // 페이지 폴트가 발생한 주소, 에러 코드를 저장할 버퍼
    char addr_buf[11];
    char err_buf[11];
	
    // 16진수 문자열로 변환
    hex32_to_str(fault_addr, addr_buf);
    hex32_to_str(error_code, err_buf);
	
    // 페이지 폴트가 발생한 주소, 에러 코드 출력
    put_string(120, 28, addr_buf, COLOR_WHITE);
    put_string(120, 48, err_buf,  COLOR_WHITE);
	
    // 백버퍼 내용을 실제 VGA로 반영
    gfx_present();
    
    for (;;)
    {
        __asm__ volatile ("hlt");
    }
}

...

// 인터럽트 초기화 함수
void interrupts_init(void)
{
	...
	
    // 페이지 폴트 인터럽트 설정 -> IDT 인덱스 0x0E에 매핑됨
    // 페이지 폴트 발생 시 page_fault_isr 호출
    // 셀렉터(GDT 코드 세그먼트) 0x08
    // type_attributes 0x8E (DPL 0, 32비트 인터럽트 게이트)
    idt_set_gate(0x0E, (uint32_t)page_fault_isr, 0x08, 0x8E);
	
    // IDT 로드
    idt_load((uint32_t)&idtp);
}

...
```

x86 예외 번호 `0x0E`에 **페이지 폴트 인터럽트 서비스 루틴을 연결**하였습니다.

페이지 폴트 핸들러에서는 **인자로 받은 주소와 에러 코드를 16진수 문자열로 붉은 화면에 출력**하도록 구현하였습니다.

```C
// 페이지 폴트 테스트 (임시)
static void test_page_fault(void)
{
    // 매핑되지 않은 주소 접근
    volatile uint32_t* p = (volatile uint32_t*)0x00400000u;
    volatile uint32_t v = *p;
    (void)v;
}

...

// 앱 작업을 위한 태스크
static void app_task(void* arg)
{
    (void)arg;
    
    // 테스트 시작 텍스트 출력
    window_put_string(g_testwin, "[PF] test start\n", COLOR_BROWN);
	
    task_sleep(100);
	
    // 페이지 폴트 발생 코드
    // test_page_fault();
	
    // 테스트 성공 메시지 출력
    window_put_string(
	    g_testwin,
	    "[PASS] No page fault occurred!\n",
	    COLOR_GREEN
	    );
	
    for (;;)
    {
        task_sleep(100);
    }
}

```

**페이지 폴트를 발생시키는 임시 코드**(`test_page_fault()`)를 구현하였습니다.
