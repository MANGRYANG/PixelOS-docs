# 변경 사항

## 코드

**idt.asm**

```Nasm
[BITS 32]

global idt_load
global timer_isr
global keyboard_isr
global mouse_isr

extern mouse_handler
extern keyboard_handler
extern timer_handler

...

;-------------------------------------------------------------------
; 마우스 IRQ12 인터럽트 서비스 루틴
;-------------------------------------------------------------------
mouse_isr:
    pusha
    call mouse_handler          ; mouse_handler 호출    
    popa
	
    ; IRQ12는 슬레이브 PIC에 연결되어 있음
    ; 따라서 IRQ12는 슬레이브 PIC(0xA0) -> 마스터 PIC(0x20) 순으로 
    ; EOI 명령 코드 전송
    mov al, 0x20
    out 0xA0, al           ; 슬레이브 PIC EOI
    out 0x20, al           ; 마스터 PIC EOI
	
    iretd

```

**PS/2 마우스 인터럽트(IRQ12) 발생 시 실행되는 ISR을 추가**하였습니다.

**IRQ12는 슬레이브 PIC 라인에 연결**되어 있기 때문에, **슬레이브 PIC → 마스터 PIC 순서로 두 칩 모두에 EOI 신호를 보내도록 작성**하였습니다.

**interrupts.h**

```C
#pragma once

#include <stdint.h>

void interrupts_init(void);         // IDT 및 PIC 설정
void irq_enable_timer(void);        // 타이머 IRQ 활성화
void irq_enable_keyboard(void);     // 키보드 IRQ 활성화
void irq_enable_mouse(void);        // 마우스 IRQ 활성화

void timer_handler(void);           // 타이머 인터럽트 핸들러
void keyboard_handler(void);        // 키보드 인터럽트 핸들러
void mouse_handler(void);           // 마우스 인터럽트 핸들러
```

마우스 인터럽트 핸들러 함수(`mouse_handler()`)를 추가하였습니다.

`timer_init()` 함수와 `keyboard_init()` 함수의 함수명을 `irq_enable_timer()`와 `irq_enable_keyboard()`로 수정하였습니다.


**interrupts.c**

```C
...

// asm 코드에서 호출되는 마우스 핸들러 정의
void mouse_handler(void)
{
    // PS/2 데이터를 읽기
    uint8_t data = inb(0x60);
    // 마우스 드라이브에 데이터 전달
    mouse_on_data(data);
}

...

// 인터럽트 초기화 함수
void interrupts_init(void)
{
	...
	
    // 마우스 인터럽트 설정(IRQ12) -> IDT 인덱스 0x2C에 매핑됨
    // 마우스 인터럽트 발생 시 mouse_isr 호출
    // 셀렉터(GDT 코드 세그먼트) 0x08
    // type_attributes 0x8E (DPL 0, 32비트 인터럽트 게이트)
    idt_set_gate(0x2C, (uint32_t)mouse_isr, 0x08, 0x8E);
    
    // IDT 로드
    idt_load((uint32_t)&idtp);
}
   
...

// 마우스 활성화 (마스크 해제)
void irq_enable_mouse(void)
{
    // 슬레이브 PIC 활성화를 위해 IRQ2 라인 마스킹 해제
    uint8_t mask_master = inb(PIC1_DATA);
    mask_master &= ~(1 << 2);
    outb(PIC1_DATA, mask_master);

    // 마우스 IRQ12 마스킹 해제
    uint8_t mask_slave = inb(PIC2_DATA);
    mask_slave &= ~(1 << 4);
    outb(PIC2_DATA, mask_slave);
}

```

**PS/2 데이터를 읽어 마우스 드라이버로 전달**하도록 `mouse_handler()` 함수를 구현하였습니다.

`interrupt_init()` 함수에서 `idt_set_gate()` 함수를 통해 **CPU가 PIC로부터 마우스 인터럽트 벡터 번호를 수신하였을 때, `mouse_isr`으로 점프**하도록 설정하였습니다.

**마우스 인터럽트를 활성화**하는 `irq_enable_mouse()` 함수를 추가하였습니다.


**mouse.h**

```C
#pragma once

#include <stdint.h>

// 버튼 비트 플래그
#define MOUSE_LEFT    0x01
#define MOUSE_RIGHT   0x02
#define MOUSE_MIDDLE  0x04

// 마우스 초기화 함수
void mouse_init(void);

// 인터럽트 핸들러에서 호출하는 마우스 데이터 전달 함수
void mouse_on_data(uint8_t data);

// 패킷(3 바이트) 완성 시 처리 함수
void mouse_on_packet(uint8_t packet[3]);

// 상태 조회
int get_mouse_x(void);
int get_mouse_y(void);
uint8_t get_mouse_buttons(void);
```

**PS/2 마우스 입력(IRQ12) 처리 모듈의 인터페이스**를 정의한 헤더 파일입니다.


**mouse.c**

```C
#include "mouse.h"
#include "../kernel/io.h"
#include "../graphics/graphics.h"

// 내부 마우스 상태
static int mouse_x = WIDTH / 2;
static int mouse_y = HEIGHT / 2;
static uint8_t mouse_buttons = 0;

// 3바이트 패킷 조립
static uint8_t packet[3];
static uint8_t packet_cycle = 0;

// 0x64 포트의 1번째 비트의 값이 해제될 때까지 대기
// 해제되었다면 0x64 포트로 명령을 전달할 수 있음
static void mouse_wait_send(void)
{
    while (inb(0x64) & 0x02);
}

// 0x64 포트의 0번째 비트의 값이 해제될 때까지 대기
// 해제되었다면 0x64 포트에서 바이트를 읽어올 수 있음
static void mouse_wait_receive(void)
{
    while (!(inb(0x64) & 0x01));
}

// 마우스 초기화 함수
void mouse_init(void)
{
    // -- Compaq 상태 설정/IRQ12 활성화 --
	
    // 명령을 보낼 수 있을 때까지 대기
    mouse_wait_send();
    // 현재 Compaq Status Byte를 포트 0x60로 내보내는 명령 전송
    outb(0x64, 0x20);   
	
    // 바이트를 읽어 올 수 있을 때까지 대기
    mouse_wait_receive();
    // 현재 Compaq Status Byte 읽기 
    uint8_t status = inb(0x60);
    // 현재 Compaq Status Byte의 비트 1을 1로 설정
    // IRQ12 활성화
    status |= 0x02;
    // 현재 Compaq Status Byte의 비트 5를 0으로 설정
    // 마우스 클럭 비활성화
    status &= ~0x20;
	
    // 명령을 보낼 수 있을 때까지 대기
    mouse_wait_send();
    // 다음 0x60에 쓸 값을 Compaq Status로 설정하는 명령 전송 
    outb(0x64, 0x60);
	
    // 명령을 보낼 수 있을 때까지 대기
    mouse_wait_send();
    // 수정된 Compaq Status Byte를 포트 0x60에 쓰기
    outb(0x60, status);
	
    // -- 보조 입력 활성화 명령 --
	
    // 명령을 보낼 수 있을 때까지 대기
    mouse_wait_send();
    // 보조 장치 활성화 명령 전송
    outb(0x64, 0xA8);  
	
    // -- PS/2 마우스 명령 --
	
    // 명령을 보낼 수 있을 때까지 대기
    mouse_wait_send();
    // 다음 0x60에 쓸 값을 마우스 명령으로 사용하도록 명령 전송
    outb(0x64, 0xD4);
    
    // 명령을 보낼 수 있을 때까지 대기
    mouse_wait_send();
    // 패킷 스트리밍 활성화 명령 전송
    outb(0x60, 0xF4);
	
    // 바이트를 읽어 올 수 있을 때까지 대기
    mouse_wait_receive();
    // 마우스로부터 ACK 응답 읽기
    inb(0x60);
}

// 인터럽트 핸들러에서 호출하는 마우스 데이터 전달 함수
void mouse_on_data(uint8_t data)
{
    if (packet_cycle == 0)
    {
        // 패킷의 첫 번째 바이트의 비트 3을 검사
        if (!(data & 0x08))
        {
            return;
        }
    }
	
    packet[packet_cycle++] = data;
	
    if (packet_cycle == 3)
    {
        packet_cycle = 0;
        mouse_on_packet(packet);
    }
}

// 3바이트 패킷이 완성되면 실행되는 함수
void mouse_on_packet(uint8_t p[3])
{
    int dx = (int8_t)p[1];
    int dy = (int8_t)p[2];
    
    if (dx < -50 || dx > 50 || dy < -50 || dy > 50)
    {
        return;
    }

    mouse_x += dx;
    // PS/2 마우스 - y축 반대 방향으로 이동
    mouse_y -= dy;

    // 화면 경계 체크
    if (mouse_x < 0) mouse_x = 0;
    if (mouse_y < 0) mouse_y = 0;
    if (mouse_x >= WIDTH)  mouse_x = WIDTH - 1;
    if (mouse_y >= HEIGHT) mouse_y = HEIGHT - 1;

    // 버튼 업데이트
    mouse_buttons = 0;
    if (p[0] & 0x01) mouse_buttons |= MOUSE_LEFT;
    if (p[0] & 0x02) mouse_buttons |= MOUSE_RIGHT;
    if (p[0] & 0x04) mouse_buttons |= MOUSE_MIDDLE;
}

// Getter 함수
int get_mouse_x(void) { return mouse_x; }
int get_mouse_y(void) { return mouse_y; }
uint8_t get_mouse_buttons(void) { return mouse_buttons; }
```

앞서 **키보드 드라이버 구현 단계**에서는 인터럽트 발생 시 **이미 데이터가 준비된 상태에서 스캔코드를 읽기만 하면 되었으므로**, PS/2 컨트롤러 상태 레지스터(`0x64`)의 OBF(`bit0`) 및 IBF(`bit1`)를 별도로 검사하지 않고 데이터 포트(`0x60`)에서 스캔코드를 읽어 처리하도록 구현하여도 문제가 발생하지 않았습니다.

반면 마우스의 경우에는 **좌표 데이터를 스트리밍 방식으로 전송하도록 설정**해야 하며, **초기화 과정에서는 컨트롤러에 명령을 전송하고 ACK 응답을 수신**해야 하므로, PS/2 컨트롤러 상태 레지스터(`0x64`)의 IBF/OBF 비트를 확인하여 송수신 타이밍을 정확히 맞출 필요가 있습니다.

이에 `mouse_wait_send()`와 `mouse_wait_receive()`를 통해 PS/2 컨트롤러 상태 레지스터의 비트 상태를 확인하도록 구현하였습니다.

`mouse_init` 함수는 **마우스 초기화 함수**로, 다음과 같은 작동 흐름으로 동작합니다.

1. IRQ12가 발생할 수 있도록 **Compaq status byte를 설정**합니다.
	- `0x64` 포트에 `0x20` 명령을 송신합니다. (Compaq status byte 반환 명령 바이트)
	- **데이터 포트(`0x60`)로 전달된 Compaq status byte**를 읽어옵니다.
	- Compaq status byte의 `bit1`을 1로 설정하여 **IRQ12를 활성화**합니다.
	- Compaq status byte의 `bit5`를 0으로 설정하여 **마우스 클럭을 비활성화**합니다.
	- `0x64` 포트에 `0x60` 명령을 송신합니다. (Compaq status byte 설정 명령 바이트)
	- **수정된 Compaq status byte를 데이터 포트(`0x60`)에 전달**합니다.

2. 보조 장치 활성화 명령(`0xA8`)을 포트 `0x64`로 전송합니다.

3. `0xD4`을 포트 `0x64`로 전송합니다. (다음으로 `0x60` 포트에 전달되는 바이트를 마우스 명령으로 사용하도록 하는 명령)

4. `0x60` 포트에 패킷 스트리밍 활성화 명령어(`0xF4`)를 전달합니다.

5. 마우스가 반환하는 ACK(`0xFA`) 응답을 확인합니다.

`mouse_on_data` 함수는 **마우스 패킷을 조립하는 함수**로, 첫 바이트의 동기 비트(`bit3`)를 확인하여 패킷의 시작점을 동기화한 뒤, 수신된 데이터를 순차적으로 저장하여 3바이트 패킷을 구성하도록 구현되었습니다.

`mouse_on_packet` 함수는 패킷의 `byte1`, `byte2`를 통해 dx/dy를 추출하고, 패킷의 `byte0`의 하위 비트(`bit0`, `bit1`, `bit2`)를 통해 마우스 버튼을 추출하여 **내부 마우스 상태를 갱신하는 함수**입니다.

`get_mouse_x`, `get_mouse_y` 함수를 통해 화면에서의 마우스 좌표를 확인하고, `get_mouse_buttons` 함수를 통해 마우스 버튼 상태를 확인할 수 있도록 Getter 함수를 구현하였습니다.

# 이슈 및 버그

## 증상

 - 마우스가 상하반전되어 움직이는 증상 발생

## 재현

 - `mouse_on_packet()`에서 패킷의 이동값을 좌표에 반영할 때, `mouse_y` 갱신 시 y축 방향을 반전하지 않고 적용

## 원인(추정)

 - PS/2 마우스 패킷의 Y 이동값은 위 방향이 양수인 반면에 그래픽 화면 좌표계(VGA)는 **아래로 갈수록 y값이 증가하는 구조**이므로, 두 좌표계가 서로 반대여서 그대로 더했을 때 상하 반전이 발생하는 것으로 추정됨

## 해결 및 우회 방법

 - `mouse_on_packet()`에서 y축 이동값을 화면 좌표계에 맞게 반전하여 적용하여 해결
