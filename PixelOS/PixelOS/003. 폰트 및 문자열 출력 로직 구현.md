# 변경 사항

## 코드

**font.h**

```C
#pragma once

#include <stdint.h>

#define WIDTH 320
#define HEIGHT 200
#define VGA_ADDR 0xA0000
#define WHITE 0X0F

#define FONT_WIDTH  8
#define FONT_HEIGHT 16

// ASCII 0~127용 폰트 배열
extern uint8_t font8x16[128][FONT_HEIGHT];

// 글자 출력 함수
void put_char(int px, int py, char ascii, uint8_t color);

// 문자열 출력 함수
void put_string(int px, int py, const char* str, uint8_t color);

#endif
```

320×200 VGA 프레임버퍼에 **8x16 크기의 비트맵 폰트(ASCII 0~127)를 출력**하기 위한 상수 및 인터페이스를 정의한 헤더 파일입니다.


**font.c**

```C
#include "font.h"

/*
 * font8x16 — ASCII 0~127 문자에 대한 8x16 Monospaced Bitmap Font Table 
 * Original source: https://github.com/hubenchang0515/font8x16  
 * License: MIT License  
 */
uint8_t font8x16[128][16] =
{
    [0]   = { 0x00, 0x00, ..., 0x00 }, //0x00

...

    [127] = { 0x00, 0x00, ..., 0x00 }, //0x7F
};

uint8_t* const vga = (uint8_t*)VGA_ADDR;

void put_char(int px, int py, char ascii, uint8_t color)
{
    // 출력할 문자의 top-left 비트의 화면 기준 인덱스
    int top_left_idx = px + (py * 320);
    // 현재 픽셀의 인덱스 값을 저장할 임시 변수
    int current_pixel_idx = 0;
    
    for(int line = 0; line < FONT_HEIGHT; ++line)
    {
        for(int i = 0; i < FONT_WIDTH; ++i)
        {   
            // 폰트 정보에서 (i, line) 픽셀에 해당하는 값이 1인 경우
            if(((font8x16[ascii][line] >> 
	            ((FONT_WIDTH - 1) - i)) & 1) == 1)
            {
                // 현재 픽셀의 화면 기준 인덱스 계산
                current_pixel_idx = 
	                (top_left_idx + i) + (WIDTH * line);
	                
                // 인자로 받은 color로 현재 픽셀의 색 변경
                vga[current_pixel_idx] = color;
            }
            
        }
    }
}

void put_string(int px, int py, const char* str, uint8_t color)
{
    int ix = px;
    for(; *str != 0x00; ++str)
    {
        put_char(ix, py, *str, color);
        ix += FONT_WIDTH;
    }
}
```

`font8x16`의 비트맵 폰트를 읽어 VGA 선형 프레임버퍼에 **8×16 픽셀**로 문자를 출력하는 로직을 작성한 코드입니다.

**비트맵 폰트는 다음 깃허브 리포지토리에서 발췌하였습니다. (MIT 라이센스)**
	https://github.com/hubenchang0515/font8x16

`put_char()` 함수를 사용하여 하나의 비트맵 문자를 화면에 출력하거나, `put_string()` 함수를 사용하여 문자열을 한번에 출력할 수 있습니다.


**kernel.c**

```C
#include <stdint.h>
#include "../font/font.h"

#define WIDTH 320
#define HEIGHT 200
#define VGA_ADDR 0xA0000
#define WHITE 0X0F

void kernel_main(void)
{
    // VGA 그래픽 메모리 시작
    uint8_t* const vga = (uint8_t*)VGA_ADDR;

    for(int i = 0; i < WIDTH*HEIGHT; ++i)
    {
        vga[i] = WHITE;  // 흰색 픽셀
    }

    put_string(8, 0, "Hello, PixelOS!", 0x00);
}
```

`font.h`의 `put_string()` 함수를 사용하여 문자열을 출력하도록 작성하였습니다.
