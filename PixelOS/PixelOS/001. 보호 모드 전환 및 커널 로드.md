# 변경 사항

## 코드

**boot.asm**

```Nasm
[BITS 16]               ; 16 bit 리얼 모드
[ORG 0x7C00]            ; 0x7C00부터 부트 섹터 시작

;-----------------------------------------------------------------
; start 레이블
;-----------------------------------------------------------------
start:
    ; 세그먼트 레지스터 초기화
    cli			            ; 초기화 중 안전을 위한 인터럽트 비활성화
    xor ax, ax		        ; ax = 0x00
    mov ds, ax		        ; ds = 0x00
    mov es, ax		        ; es = 0x00
    mov ss, ax		        ; ss = 0x00
    mov sp, 0x7C00	        ; 스택 포인터를 부트 섹터 시작점으로 설정
    sti			            ; 인터럽트 활성화

    mov si, MSG_HELLO       ; 문자열 시작 주소 저장
    call print_loop         ; 문자열 출력

    call load_kernel        ; 커널 로드

;------------------------------------------------------------------
; 외부 코드 포함
;------------------------------------------------------------------
%include "boot/print_string.asm"
%include "boot/load_kernel.asm"

;------------------------------------------------------------------
; 데이터
;------------------------------------------------------------------
MSG_HELLO db 'Hello, PixelOS Booted from HDD!', 0x0A, 0

;------------------------------------------------------------------
; 부트섹터 패딩 및 시그니처
;------------------------------------------------------------------
times 510-($-$$) db 0   ; 부트섹터 크기를 512 byte로 맞춤 설정
dw 0xAA55               ; BIOS 부트 시그니처 : 마지막 2 byte
```

`load_kernel.asm` 파일을 포함하여 `load_kernel` 루틴을 추가하고, 부트 코드에서 해당 루틴을 호출하도록 수정하였습니다.


**load_kernel.asm**

```Nasm
;------------------------------------------------------------------
; 커널 로더
;------------------------------------------------------------------
load_kernel:
    mov si, MSG_LOAD_KERNEL
    call print_loop

    ; 커널 임시 로드 (0x8000)
    mov ah, 0x42            ; 확장된 디스크 읽기 모드 설정
    mov dl, [boot_drive]    ; 드라이브 번호 (첫 번째 하드디스크)
    mov si, kernel_dap      ; DAP 구조체 주소
    int 0x13                ; BIOS Interrupt (디스크 I/O 서비스)
    jc .load_error          ; 에러 발생 시 load_error로 점프

    call graphic_mode       ; 화면 모드 전환 (그래픽 모드)

    jmp entry_PM            ; 보호 모드 진입

.load_error:
    mov si, MSG_LOAD_KERNEL_TEMP_ERROR
    call print_loop

    jmp $

;------------------------------------------------------------------
; DAP 구조체
;------------------------------------------------------------------
kernel_dap:
    db 0x10                 ; DAP 구조체 크기 : 16바이트
    db 0                    ; 예약된 공간
    dw 4                    ; 읽을 섹터 수
    dw 0x8000               ; BX : 오프셋
    dw 0                    ; ES : 세그먼트
    dd 1                    ; LBA 하위 32비트
    dd 0                    ; LBA 상위 32비트

;------------------------------------------------------------------
; 외부 코드 포함
;------------------------------------------------------------------
%include "boot/video_mode.asm"
%include "boot/switch_pm.asm"

;------------------------------------------------------------------
; 데이터
;------------------------------------------------------------------
MSG_LOAD_KERNEL db "Loading kernel into main memory....", 0x0A, 0
MSG_LOAD_KERNEL_TEMP_ERROR db "Failed to load kernel into temporary space", 0x0A, 0

boot_drive db 0x80          ; 첫 번째 하드디스크

;------------------------------------------------------------------
; PM 작업 코드
;------------------------------------------------------------------
[BITS 32]

PROTECTEDMODE:
    mov ax, DATA_SEG
    mov ds, ax
    mov es, ax
    mov fs, ax
    mov gs, ax
    mov ss, ax
    mov esp, 0x0009FC00

    ; 커널 재배치
    mov esi, 0x00008000     ; source (0x8000)
    mov edi, 0x00100000     ; destination (0x100000)
    mov ecx, 512            ; 2048 bytes / 4 bytes = 512
    cld
    rep movsd               ; 4바이트씩 복사

    jmp CODE_SEG:0x00100000 ; 커널 진입

```

16비트 리얼 모드에서 **BIOS가 커널을 로드하는 루틴을 기술하는 코드**입니다.
**커널 로드 루틴**은 다음과 같습니다.
1. 먼저 리얼 모드에서 접근 가능한 **임시 주소**인 `0x0000:0x8000`로 커널을 읽어옵니다.
2. 화면 모드를 **그래픽 모드로 전환**한 뒤 **보호 모드로 진입**합니다.
3. 임시로 로드한 커널 코드를 `0x00100000(1MB)`으로 복사합니다.
4. `0x00100000`으로 점프하여 커널을 시작합니다.

커널을 메모리에 로드하기 위하여 **DAP(Disk Address Packet)** 구조체를 정의한 뒤 BIOS 확장 디스크 읽기 서비스에 전달하는 방식을 사용하였습니다.

리얼 모드에서는 커널을 배치하기에 안전한 메모리 주소인 `0x00100000(1MB)`에 **직접적으로 접근하기 어려우므로,** 리얼 모드에서 접근할 수 있는 **임시 주소에 커널을 로드한 뒤 보호 모드로 진입하여 해당 커널 코드를 복사**하는 방식으로 커널을 로드하였습니다.


**switch_pm**

```Nasm
;-------------------------------------------------------------------
; Protected Mode 진입
;-------------------------------------------------------------------
entry_PM:
    cli                     ; 인터럽트 비활성화
    lgdt [gdt_descriptor]   ; GDT 로드
    call enable_a20         ; A20 활성화
    mov eax, cr0
    or al, 1
    mov cr0, eax            ; CR0 레지스터의 PE bit를 1로 설정

    jmp CODE_SEG:PROTECTEDMODE

;-------------------------------------------------------------------
; A20 활성화
;-------------------------------------------------------------------
enable_a20:
    in al, 0x92
    or al, 0x02        ; bit1 = A20
    and al, 0xFE       ; bit0 = reset disable
    out 0x92, al
    ret

;-------------------------------------------------------------------
; 외부 코드 포함
;-------------------------------------------------------------------
%include "boot/gdt.asm"
```

**16비트 리얼 모드에서 32비트 보호 모드로 진입하는 루틴을 기술**한 코드입니다.
보호 모드 진입 루틴은 다음과 같습니다.
1. **GDT(Global Descriptor Table)**를 CPU에 등록합니다.
2. 1MB 이상 주소 접근을 위해 **A20 라인을 활성화**합니다.
3. `CR0` 레지스터의 **PE(Protection Enable) 비트를 1으로 설정**하여 보호 모드를 켭니다.
4. **32비트 코드 시작 지점으로 점프**합니다.


**gdt.asm**

```Nasm
;-------------------------------------------------------------------
; GDT 정의
;-------------------------------------------------------------------
gdt_start:
    dq 0x0000000000000000       ; Null Descriptor
    dq 0x00CF9A000000FFFF       ; Code segment
    dq 0x00CF92000000FFFF       ; Data segment
gdt_end:

gdt_descriptor:
    dw gdt_end - gdt_start - 1  ; GDT 크기
    dd gdt_start                ; GDT 시작 주소

CODE_SEG equ 0x08
DATA_SEG equ 0x10
```

보호 모드로 진입하기 위해 필요한 **GDT(Global Descriptor Table)를 정의한 코드**입니다.
`lgdt [gdt_descriptor]`과 같이 사용하여 GDT의 **base**와 **limit**을 CPU에 등록할 수 있습니다.


**video_mode.asm**

```Nasm
;-------------------------------------------------------------------
; 텍스트 모드로 전환
;-------------------------------------------------------------------
text_mode:
    mov ah, 0x00
    mov al, TEXT_MODE_NUM
    int 0x10
    ret
    
;-------------------------------------------------------------------
; 그래픽 모드로 전환
;-------------------------------------------------------------------
graphic_mode:
    mov ah, 0x00
    mov al, GRAPHIC_MODE_NUM
    int 0x10
    ret

TEXT_MODE_NUM equ 0x03
GRAPHIC_MODE_NUM equ 0x13
```

16비트 리얼 모드에서 동작하는 **비디오 모드 전환 루틴을 기술하는 코드**입니다.
BIOS 비디오 서비스(`int 0x10`)를 호출 시 `ah` 레지스터의 값을 `0x00`으로 설정하면 **비디오 모드 설정 기능(Set Video Mode)을 사용할 수 있습니다.**
이 때 `al` 레지스터에 지정한 모드 번호에 따라 전환되는 비디오 모드가 결정됩니다.

# 이슈 및 버그

## 증상

 - 커널 로드 과정에서 **트리플 폴트(Triple Fault)가 발생하며 부팅 실패**

## 재현

 - DAP 구조체의 `ES(세그먼트)`를 `0xFFFF`, `BX`를 `0x0010`으로 설정하여 **리얼 모드에서 `0x00100000`에 커널을 직접 로드하도록 시도**

## 원인(추정)

 - 리얼 모드에서 `0x00100000(1MB)` 이상의 영역에 직접 접근할 시 **주소 래핑(wrap-around)** 현상이 발생하여 `0x0000` 영역으로 접근되는 것으로 추정됨

## 해결 및 우회 방법

 - 커널을 리얼 모드에서 접근 가능한 범위의 **임시 주소(`0x8000`)에 로드**
 - **A20 라인 활성화 및 보호 모드 진입을 수행**한 후, 커널을 `0x00100000`에 복사하여 재배치하는 방식을 사용
