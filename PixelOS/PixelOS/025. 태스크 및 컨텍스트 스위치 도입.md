# 변경 사항

## 코드

**task_switch.asm**

```Nasm
[BITS 32]
global task_switch

; void task_switch(uint32_t** old_esp, uint32_t* new_esp)
task_switch:
    cli
	
    mov eax, [esp+4]    ; old_esp (현재 태스크의 ESP를 저장할 변수의 주소)
    mov edx, [esp+8]    ; new_esp (다음 태스크의 ESP를 저장할 변수의 주소)
	
    ; 현재 스택의 레지스터 정보 보존을 위한 push
    push ebp
    push ebx
    push esi
    push edi
	
    ; 현재 스택 포인터 저장
    mov [eax], esp
	
    ; 태스크 전환
    mov esp, edx
	
    ; 전환된 태스크의 레지스터 정보 pop
    pop edi
    pop esi
    pop ebx
    pop ebp
	
    sti
    ret
```

**현재 태스크의 실행 컨텍스트(레지스터, 스택 포인터)를 저장**하고 **다음 태스크의 스택 포인터로 교체하여 실행을 이어주는 컨텍스트 스위칭 루틴**을 구현한 어셈블리 파일입니다.

**컨텍스트 스위칭**은 다음과 같은 흐름으로 수행됩니다.

1. 인터럽트로 인한 상태 변화 방지를 위해 `cli` 명령으로 인터럽트를 비활성화합니다.
2. 현재 태스크의 **레지스터 값을 스택에 저장한 뒤, `ESP`를 `old_esp`에 보관**합니다.
3. **다음에 실행할 태스크의 스택 포인터(`new_esp`)를 `ESP`에 로드**하여 컨텍스트를 전환합니다.
4. **전환된 스택에서 레지스터 값을 복원**하고, **`ret` 명령을 통해 다음 태스크의 실행 지점으로 복귀**합니다.


**task.h**

```C
#pragma once
#include <stdint.h>
#include <stdbool.h>
#include <stddef.h>

// 인자가 없고 반환형이 void인 함수 포인터 타입의 별칭 정의
typedef void (*task_fn)(void*);

// Task 구조체 정의
typedef struct Task {
    uint32_t* esp;          // 저장된 스택 포인터
    uint8_t* stack;         // 태스크 전용 스택 메모리 바닥 주소
    size_t stack_size;      // 태스크 전용 스택 크기
    task_fn fn;             // 태스크 시작 시점에서 실행될 함수 포인터
    void* arg;              // 해당 함수에 넘겨줄 인자
    bool alive;             // 태스크 실행 대상 여부
} Task;

// 태스크 테이블 초기화 함수
void task_init(void);
// 새 태스크 생성 함수
int task_create(task_fn fn, void* arg, size_t stack_size);

// 첫 태스크로 전환하는 함수
void task_start(void);
// 태스크가 CPU 점유를 양보하는 함수 (협력 스케줄링)
void task_yield(void);
// 태스크 종료 함수
__attribute__((noreturn)) void task_exit(void);
```

**태스크를 실행 컨텍스트 단위로 관리**하기 위해 필요한 인터페이스를 선언한 헤더 파일입니다.

태스크를 실행 컨텍스트 단위로 관리할 수 있도록 태스크 구조체를 정의하였습니다.
태스크 구조체의 멤버는 다음과 같습니다.

- `esp` : 컨텍스트 스위칭 시 저장 및 복원되는 **스택 포인터(`ESP`)**입니다.
- `stack` : 태스크가 사용할 **전용 스택 메모리의 시작 주소(바닥 주소)**입니다.
- `stack_size` : 전용 스택 메모리의 **크기**입니다.
- `fn` : 태스크가 최초 실행될 때 진입하는 **엔트리 함수 포인터**입니다.
- `arg` : 엔트리 함수(`fn`)에 전달할 **인자 포인터**입니다.
- `alive` : 태스크가 현재 **스케줄링 대상인지 여부**를 나타내는 플래그입니다.


**task.c**

```C
#include "task.h"
#include "heap.h"

#define MAX_TASKS 4

extern void task_switch(uint32_t** old_esp, uint32_t* new_esp);

// 태스크를 저장할 배열(테이블)
static Task g_tasks[MAX_TASKS];
static int g_count = 0;
static int g_current = -1;

// 메인 컨텍스트 저장용 변수
static uint32_t* g_main_esp = 0;

// 트램폴린에서 현재 태스크를 알아내기 위한 포인터
static Task* g_current_task = 0;

// 유휴 상태 루프
__attribute__((noreturn))
static void idle_loop(void)
{
    for (;;) __asm__ volatile("hlt");
}

__attribute__((noreturn))
static void task_trampoline(void)
{
    Task* t = g_current_task;
    if (!t || !t->fn) idle_loop();
	
    t->fn(t->arg);
    task_exit();
}

// 태스크 테이블 초기화 함수
void task_init(void)
{
    for (int i = 0; i < MAX_TASKS; ++i) {
        g_tasks[i].esp = 0;
        g_tasks[i].stack = 0;
        g_tasks[i].stack_size = 0;
        g_tasks[i].fn = 0;
        g_tasks[i].arg = 0;
        g_tasks[i].alive = false;
    }
    g_count = 0;
    g_current = -1;
    g_main_esp = 0;
    g_current_task = 0;
}

// 새 태스크가 처음 생성될 때 실행될 태스크의 초기 스택 구조 설정 함수
static uint32_t* build_initial_stack(uint8_t* stack, size_t stack_size)
{
    // 새 태스크의 시작 주소(top)
    uint32_t* sp = (uint32_t*)(stack + stack_size);
    
    // task_switch가 new esp로 바꾼 뒤:
    // pop edi, pop esi, pop ebx, pop ebp, ret 순서로 실행됨
    *(--sp) = (uint32_t)task_trampoline; // ret로 점프할 주소
    *(--sp) = 0; // ebp
    *(--sp) = 0; // ebx
    *(--sp) = 0; // esi
    *(--sp) = 0; // edi
	
    return sp;
}

// 새 태스크 생성 함수
int task_create(task_fn fn, void* arg, size_t stack_size)
{
    // 실행할 엔트리 함수가 없는 경우 실패
    if (!fn) { return -1; }
    // 태스크 수용 범위를 넘어선 경우 실패
    if (g_count >= MAX_TASKS) { return -1; }
	
    // 최소 스택 크기 보장(4KB)
    if (stack_size < 4096) stack_size = 4096;
	
    // 새 태스크 생성 후 태스크 테이블에 등록
    Task* t = &g_tasks[g_count];
    // 스택 크기 만큼의 공간을 kmalloc하여 스택 바닥 주소를 보관
    t->stack = (uint8_t*)kmalloc_aligned(stack_size, 16);
    // 할당 실패 시 태스크 생성 실패
    if (!t->stack) { return -1; }
	
    // 태스크 구조체 정보 설정
    t->stack_size = stack_size;
    t->fn = fn;
    t->arg = arg;
    t->alive = true;
    // 스택 기본 구조 생성
    t->esp = build_initial_stack(t->stack, t->stack_size);
	
    return g_count++;
}

// 다음에 실행할 alive 상태의 태스크 탐색 (Round-Robin 방식)
static int pick_next(int from)
{
    // 태스크 테이블이 비어 있는 경우 실패
    if (g_count == 0) { return -1; }
	
    for (int i = 1; i <= g_count; ++i)
    {
        int idx = (from + i) % g_count;
        if (g_tasks[idx].alive)
        {
            return idx;
        }
    }
    return -1;
}

// 첫 번째 태스크로 컨텍스트 스위치하는 함수
void task_start(void)
{
    // 실행할 첫 태스크 탐색
    int next = pick_next(-1);
    // 실행할 태스크가 없는 경우 유휴 상태 돌입
    if (next < 0) { idle_loop(); }
	
    g_current = next;
    g_current_task = &g_tasks[g_current];
	
    // 해당 태스크로 컨텍스트 스위칭
    task_switch(&g_main_esp, g_tasks[g_current].esp);
	
    idle_loop();
}

// 태스크가 CPU 점유를 양보하는 함수 (협력 스케줄링)
void task_yield(void)
{
    // 태스크가 1개 이하인 경우 양보하지 않음
    if (g_count <= 1) { return; }
	
    // 다음으로 실행 가능한 태스크 탐색
    int next = pick_next(g_current);
    // 다음으로 실행 가능한 태스크가 현재 태스크와 같거나 없는 경우 양보 X
    if (next < 0 || next == g_current) { return; }
	
    int prev = g_current;
    g_current = next;
    g_current_task = &g_tasks[g_current];
	
    // 해당 태스크로 컨텍스트 스위칭
    task_switch(&g_tasks[prev].esp, g_tasks[g_current].esp);
}

// 태스크 종료 함수
__attribute__((noreturn))
void task_exit(void)
{
    // 태스크의 alive 상태 false로 변경
    if (g_current >= 0 && g_current < g_count)
    {
        g_tasks[g_current].alive = false;
    }
	
    // 다음으로 실행할 태스크 탐색
    int next = pick_next(g_current);
    // 실행 가능한 태스크가 없는 경우 유휴 상태 돌입
    if (next < 0) { idle_loop(); }
	
    int prev = g_current;
    g_current = next;
    g_current_task = &g_tasks[g_current];
	
    // 해당 태스크로 컨텍스트 스위칭
    task_switch(&g_tasks[prev].esp, g_tasks[g_current].esp);
	
    idle_loop();
}
```

전용 스택 메모리를 가지는 **태스크를 생성 및 관리**하고, **태스크 간의 컨텍스트 스위칭을 수행**하는 협력 스케줄링 모듈입니다.

정적 태스크 테이블(`g_tasks`)에 태스크를 저장하며, `alive` 상태인 태스크들을 **라운드 로빈(Round-Robin) 방식으로 선택하여 실행**하도록 구현하였습니다.

**실제 컨텍스트 스위칭**은 **외부 어셈블리 루틴인 `task_switch`를 호출해 수행**하며, 현재 태스크의 `ESP`를 저장한 뒤 다음 태스크의 `ESP`를 복원하는 방식으로 전환합니다.

모듈에서 제공하는 주요 인터페이스는 다음과 같습니다.

- `task_init` : 태스크 테이블(`g_tasks`)과 관련 전역 상태를 초기화합니다.
- `task_create` : 태스크 전용 스택을 할당하고, 초기 스택 프레임을 구성하여 **새 태스크를 등록**합니다.
- `task_start` : **실행 가능한 첫 태스크로 최초 컨텍스트 스위칭을 수행**합니다.
- `task_yield` : **다음으로 실행 가능한 태스크를 선택**하여 **라운드 로빈 방식**으로 컨텍스트 스위칭을 수행합니다.
- `task_exit` : **현재 태스크를 종료 처리(`alive = false`)한 뒤 다음 태스크로 전환**합니다.

**kernel.c**

```C
...

// UI 작업을 위한 태스크
static void ui_task(void* arg)
{
    (void)arg;
	
    uint32_t last_tick = g_timer_ticks;
	
    for (;;) {
        // 프레임 경계 대기
        while (g_timer_ticks == last_tick) {
            asm volatile("hlt");
        }
        last_tick = g_timer_ticks;
		
        // 이벤트 처리
        Event ev;
        while (event_pop(&ev)) {
            if (ev.type == EV_MOUSE_MOVE) {
                cursor_set_pos(ev.mouse_move.x, ev.mouse_move.y);
            } else if (ev.type == EV_KEY) {
                if (ev.key.pressed && ev.key.ascii && g_testwin) {
                    window_put_char(
	                    g_testwin, ev.key.ascii, COLOR_BLACK
	                );
                }
            }
        }
		
        compositor_compose();
        task_yield();
    }
}

// 앱 작업을 위한 태스크
static void app_task(void* arg)
{
    (void)arg;
	
    for (;;) {
            window_put_char(g_testwin, '.', COLOR_RED);
        }
		
        task_yield();
    }
}

void kernel_main(void)
{   
    ...
	
    task_init();
    task_create(ui_task, 0, 32768);
    task_create(app_task, 0, 32768);
	
    task_start();
}
```

기존의 프레임 루프 기반 스케줄러 대신, `task_create()`로 UI 및 앱 작업을 **독립 태스크**로 생성하고 `task_start()`로 실행하도록 수정하였습니다. (기존 스케줄러 코드는 제거)

각 태스크는 작업을 수행한 후 `task_yield()`를 호출해 CPU를 양보하며, **협력 스케줄링 방식으로 번갈아 실행**합니다.

**UI 태스크(`ui_task`)는 타이머 틱을 기준으로 프레임 경계를 맞추어 이벤트 처리 및 합성(`compositor_compose`) 작업을 수행**하고, **애플리케이션 태스크(`app_task`)는 테스트 윈도우에 점을 출력하는 작업을 수행**하도록 분리하였습니다.

# 이슈 및 버그

## 증상

 - 새로 생성한 태스크로 컨텍스트 스위칭 시, **무한 루프(`hlt`) 상태로 진입**

## 재현

 - `task_create()`로 태스크를 생성할 때, `build_initial_stack`을 호출하지 않아 **초기 스택 프레임을 구성하지 않은 상태**에서 `task_start()`를 호출

## 원인(추정)

 - `task_switch`에서는 `pop edi` → `pop esi` → `pop ebx` → `pop ebp` → `ret` 순서로 레지스터 복원 및 복귀 작업을 수행함.
 - 따라서 새 태스크의 스택에는 위 작업에서 읽을 **더미 레지스터 슬롯과 반환 주소**가 미리 배치되어 있어야 하나, 태스크 생성 시 초기 스택 프레임을 구성하지 않아 스택에 해당 값들이 없어 문제가 발생한 것으로 추정됨.

## 해결 및 우회 방법

 - **초기 스택 프레임을 구성**하는 함수(`build_initial_stack`)를 구현 및 적용하고, 태스크 생성 시 `task_switch`의 복원 순서에 맞게 스택 프레임이 구성되도록 수정하여 해결하였음.
