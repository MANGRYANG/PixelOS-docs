# 변경 사항

## **리팩토링**


**syscall.h**

```C
#pragma once
#include <stdint.h>

// 앱/커널 공용 시스템 콜 번호 정의
typedef enum SyscallNum
{
    SYS_DEBUG_PUTS = 1,
    SYS_GET_TICKS  = 2,
    SYS_KEY_DOWN   = 3,
    SYS_YIELD      = 4,
} SyscallNum;
```

공용으로 사용하는 시스템 콜 번호 정의를 `syscall.h` 파일로 분리하였습니다.


**app/api.h**

```C
#pragma once

#include <stdint.h>
#include "../kernel/syscall.h"

// ring3 앱에서 사용할 시스템 콜 API
__attribute__((section(".usertext"), always_inline))
static inline int app_debug_puts(const char* s)
{
    uint32_t ret;
    __asm__ volatile (
        "int $0x80"
        : "=a"(ret)
        : "a"(SYS_DEBUG_PUTS), "c"((uint32_t)s), "d"(0), "b"(0)
        : "memory"
    );
    return (int)ret;
}

__attribute__((section(".usertext"), always_inline))
static inline uint32_t app_get_ticks(void)
{
    uint32_t ticks;
    __asm__ volatile (
        "int $0x80"
        : "=a"(ticks)
        : "a"(SYS_GET_TICKS), "c"(0), "d"(0), "b"(0)
        : "memory"
    );
    return ticks;
}

__attribute__((section(".usertext"), always_inline))
static inline int app_key_down(uint8_t key)
{
    uint32_t ret;
    __asm__ volatile (
        "int $0x80"
        : "=a"(ret)
        : "a"(SYS_KEY_DOWN), "c"((uint32_t)key), "d"(0), "b"(0)
        : "memory"
    );
    return (int)ret;
}

__attribute__((section(".usertext"), always_inline))
static inline void app_yield(void)
{
    __asm__ volatile (
        "int $0x80"
        :
        : "a"(SYS_YIELD), "c"(0), "d"(0), "b"(0)
        : "memory"
    );
}
```

`syscall.h`에서 정의된 시스템 콜 넘버에 대한 API를 `app/api.h`으로 분리하였습니다.

권한 문제로 인한 페이지 폴트 발생을 방지하기 위해 `.usertext` 섹션에 위치하도록 설정하였습니다.
