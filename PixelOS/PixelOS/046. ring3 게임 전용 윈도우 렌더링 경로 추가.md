게임 윈도우 관련 시스템 콜 넘버를 추가하였습니다.
(`SYS_GAME_CLEAR`, `SYS_GAME_FILL_RECT`, `SYS_PRESENT`, `SYS_GAME_GET_SIZE`)


**app/api.h**

```C
...

__attribute__((section(".usertext"), always_inline))
static inline void app_game_clear(uint8_t color)
{
    __asm__ volatile (
        "int $0x80"
        :
        : "a"(SYS_GAME_CLEAR), "c"((uint32_t)color), "d"(0), "b"(0)
        : "memory"
    );
}

__attribute__((section(".usertext"), always_inline))
static inline void app_game_fill_rect(
    uint16_t x,
    uint16_t y,
    uint16_t w,
    uint16_t h,
    uint8_t color
)
{
    uint32_t pos = ((uint32_t)y << 16) | x;
    uint32_t size = ((uint32_t)h << 16) | w;
	
    __asm__ volatile (
        "int $0x80"
        :
        : "a"(SYS_GAME_FILL_RECT),
	      "c"(pos),
	      "d"(size),
	      "b"((uint32_t)color)
        : "memory"
    );
}

__attribute__((section(".usertext"), always_inline))
static inline void app_present(void)
{
    __asm__ volatile (
        "int $0x80"
        :
        : "a"(SYS_PRESENT), "c"(0), "d"(0), "b"(0)
        : "memory"
    );
}

__attribute__((section(".usertext"), always_inline))
static inline uint32_t app_game_get_size(void)
{
    uint32_t ret;

    __asm__ volatile (
        "int $0x80"
        : "=a"(ret)
        : "a"(SYS_GAME_GET_SIZE), "c"(0), "d"(0), "b"(0)
        : "memory"
    );
	
    return ret;
}
```

**게임 윈도우 렌더링 관련 API** 및 **게임 윈도우의 크기를 가져오는 API**를 정의하였습니다.

권한 문제로 인한 페이지 폴트 발생을 방지하기 위해 `.usertext` 섹션에 위치하도록 설정하였습니다.

**game_window.c**

```C
#include <stdint.h>

#include "game_window.h"
#include "../font/font.h"
#include "../graphics/color.h"
#include "../graphics/compositor.h"

static Window* g_gamewin = 0;

#define GAME_TITLE_HEIGHT (FONT_HEIGHT + 2)

// 게임 윈도우 생성 함수
Window* kernel_game_init(void)
{
    g_gamewin = wm_create_window(
	    80, 30, 220, 150, COLOR_BLACK, COLOR_BLUE, "GAME"
    );
	
    return g_gamewin;
}

void kernel_game_clear(uint8_t color)
{
    if (!g_gamewin || !g_gamewin->in_use)
    {
        return;
    }
	
    const int client_x0 = 1;
    const int client_y0 = GAME_TITLE_HEIGHT + 1;
    const int client_x1 = g_gamewin->width - 1;
    const int client_y1 = g_gamewin->height - 1;
	
    for (int y = client_y0; y < client_y1; ++y)
    {
        for (int x = client_x0; x < client_x1; ++x)
        {
            g_gamewin->buffer[y * g_gamewin->stride + x] = color;
        }
    }
}

void kernel_game_fill_rect(int x, int y, int w, int h, uint8_t color)
{
    if (!g_gamewin || !g_gamewin->in_use)
    {
        return;
    }
	
    if (w <= 0 || h <= 0)
    {
        return;
    }
	
    // 게임 렌더링 구역의 좌표로 변경
    const int origin_x = 1;
    const int origin_y = GAME_TITLE_HEIGHT + 1;
	
    int x0 = origin_x + x;
    int y0 = origin_y + y;
    int x1 = x0 + w;
    int y1 = y0 + h;
	
    const int clip_x0 = 1;
    const int clip_y0 = GAME_TITLE_HEIGHT + 1;
    const int clip_x1 = g_gamewin->width - 1;
    const int clip_y1 = g_gamewin->height - 1;
	
    if (x0 < clip_x0) x0 = clip_x0;
    if (y0 < clip_y0) y0 = clip_y0;
    if (x1 > clip_x1) x1 = clip_x1;
    if (y1 > clip_y1) y1 = clip_y1;
	
    if (x0 >= x1 || y0 >= y1)
    {
        return;
    }
	
    for (int py = y0; py < y1; ++py)
    {
        for (int px = x0; px < x1; ++px)
        {
            g_gamewin->buffer[py * g_gamewin->stride + px] = color;
        }
    }
}

void kernel_game_present(void)
{
    compositor_compose();
}

// 게임을 그릴 수 있는 영역의 크기를 가져오는 함수
uint32_t kernel_game_get_size(void)
{
    if (!g_gamewin || !g_gamewin->in_use)
    {
        return 0;
    }
	
    const int title_h = FONT_HEIGHT + 2;
	
    uint16_t client_w = 0;
    uint16_t client_h = 0;
	
    if (g_gamewin->width > 2)
    {
        client_w = (uint16_t)(g_gamewin->width - 2);
    }
	
    if (g_gamewin->height > title_h + 2)
    {
        client_h = (uint16_t)(g_gamewin->height - title_h - 2);
    }
	
    return ((uint32_t)client_h << 16) | client_w;
}
```

게임 윈도우 관련 함수를 정의한 파일(`game_window.h / game_window.c`)을 생성하였습니다.

각 함수의 역할은 다음과 같습니다.

- `Window* kernel_game_init(void)`
  게임 윈도우를 생성하여 윈도우 포인터를 반환하는 함수

- `void kernel_game_clear(uint8_t color)`
  게임 윈도우를 인자로 받은 색으로 채우는 함수

- `void kernel_game_fill_rect(int x, int y, int w, int h, uint8_t color)`
  게임 윈도우의 **클라이언트 좌표 (x, y)**에 **w * h 크기의 사각형**을 그리는 함수

- `void kernel_game_present(void)`
  화면을 합성하는 함수 (`compositor_compose()` 호출)

- `uint32_t kernel_game_get_size(void)`
  윈도우 프레임 오프셋을 제외한 **클라이언트 영역**의 좌표계를 산출하여, 상위 16비트에는 높이를, 하위 16비트에는 너비를 패킹하여 반환하는 함수

**kernel.c**

```C
...

#include "../kernel/game_window.h"

...

// 게임 화면을 렌더링할 윈도우
static Window* g_gamewin = 0;

...


// 유저 모드에서 실행할 임시 함수 (usertext 섹션)
__attribute__((section(".usertext"), noreturn))
void user_test_main(void)
{
    // 시작 위치
    int curr_x = 50, curr_y = 50;
	
    // 시작 방향
    int x_direction = 1, y_direction = 1;
	
    const int ball_w = 5;
    const int ball_h = 5;
	
    uint32_t size = app_game_get_size();
    uint16_t game_w = (uint16_t)(size & 0xFFFF);
    uint16_t game_h = (uint16_t)(size >> 16);
	
    for (;;)
    {
        // 게임 화면 초기화
        app_game_clear(COLOR_BLACK);
		
        // 현재 위치에 ball 렌더링
        app_game_fill_rect(curr_x, curr_y, ball_w, ball_h, 10);
		
        // 화면 렌더링
        app_present();
		
        // ball 위치 갱신
        curr_x += x_direction;
        curr_y += y_direction;
		
        // ball의 x 좌표가 0 이하인 경우 x 방향 반전
        if (curr_x <= 0)
        {
            curr_x = 0;
            x_direction = 1;
        }
        // ball의 x 좌표가 게임 윈도우의 너비 이상인 경우 x 방향 반전
        else if (curr_x + ball_w >= game_w)
		
        {
            curr_x = game_w - ball_w;
            x_direction = -1;
        }
		
        // ball의 y 좌표가 0 이하인 경우 y 방향 반전
        if (curr_y <= 0)
        {
            curr_y = 0;
            y_direction = 1;
        }
        // ball의 y 좌표가 게임 윈도우의 높이 이상인 경우 y 방향 반전
        else if (curr_y + ball_h >= game_h)
        {
            curr_y = game_h - ball_h;
            y_direction = -1;
        }
	    app_sleep(1);
    }
}

void kernel_main(void)
{
	...
	
	// 게임 전용 윈도우 생성
	g_gamewin = kernel_game_init();
	
	...
}
```

게임 전용 윈도우의 주소를 저장할 포인터 변수를 선언하였습니다. (`Window* g_gamewin`)

`kernel_main` 함수에서 `kernel_game_init()` 함수를 통해 게임 전용 윈도우를 생성하였습니다.

`user_test_main(void)` 함수(`Ring3`)에서 게임 전용 윈도우에 `ball`을 렌더링하고, 사용자 영역의 경계에 닿을 때마다 방향을 바꿔 이동하는 단순한 로직을 작성하였습니다.

# 이슈 및 버그

## 증상

 - 유저 함수(Game)에서 **ball이 지정된 게임 렌더링 구역을 벗어나는 현상** 발생

## 재현

 - `kernel/game_window.c` 경로의 `kernel_game_fill_rect` 함수에서 인자로 받는 **x/y 좌표를 게임 렌더링 구역에서의 좌표로 변환하는 로직을 제거**하여 재현 가능

## 원인(추정)

 - **유저 모드 앱**은 **(0, 0)을 게임 화면의 왼쪽 위로 인식**하는 반면 **커널의 렌더링 함수**는 이를 **타이틀 바를 포함한 윈도우 전체의 왼쪽 위로 인식**하여 처리해 해당 버그가 발생하였습니다.

## 해결 및 우회 방법

 - **게임 전용 윈도우 내부에 사용자 영역을 정의**하고 모든 렌더링 인자에 **오프셋을 더하는 방식**을 사용하여 해결하였습니다.
